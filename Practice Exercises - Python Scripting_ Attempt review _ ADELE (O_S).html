      x=y=0

        if x == goal: return True
        x=x+1

        y=y+1

    def CommonWord(string):
      import re
      words=string.lower()
      words= re.sub(r'[^\w\s]','',words)
      words=words.split(" ")
      counts={}
      biggest=0
      for word in words:
        if word not in counts:
          counts[word] =0
        counts[word]+=1
      for x in counts:
        if counts[x] > biggest:
          biggest=counts[x]
      return(biggest)

      direction=[]
      distance=[]

      horizontal=0

        distance.append(((moves[x]).split(" ")[1]))
        if direction[x]=="UP":
          vertical = vertical + int(distance[x])
        if direction[x]=="DOWN":
          vertical = vertical - int(distance[x])
        if direction[x]=="LEFT":
          horizontal = horizontal - int(distance[x])
        if direction[x]=="RIGHT":
          horizontal = horizontal + int(distance[x])
        x=x+1
      return(abs(vertical+horizontal))
